前中后序遍历是什么：

前序：根 – 左 – 右

中序：左 – 根 – 右

后序：左 – 右 – 根

二叉树的前中后序遍历（迭代版）（利用栈）

前序

/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     public int val;
 *     public TreeNode left;
 *     public TreeNode right;
 *     public TreeNode(int val=0, TreeNode left=null, TreeNode right=null) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
public class Solution {
    public IList<int> PreorderTraversal(TreeNode root) {
        List<int> ans = new List<int>();
        Stack<TreeNode> st = new Stack<TreeNode>();
        if (root == null) return ans;
        st.Push(root);
        while (st.Count != 0){
            TreeNode node = st.Pop();
            ans.Add(node.val);
            if (node.right != null) st.Push(node.right);
            if (node.left != null) st.Push(node.left);
        }
        return ans;
    }
}

中序

public class Solution {
    public IList<int> InorderTraversal(TreeNode root) {
        List<int> ans = new();
        Stack<TreeNode> temp = new();
        TreeNode t = root;
        while (t != null || temp.Count != 0){
            if (t != null){
                temp.Push(t);
                t = t.left;
            }
            else{
                t = temp.Pop();
                ans.Add(t.val);
                t = t.right;
            }
        }   
        return ans;
    }
}

        tmp.Push(root);
        TreeNode t;

            rans.Add(t.val);
            if (t.left != null) tmp.Push(t.left);
            if (t.right != null) tmp.Push(t.right);
        }
        rans.Reverse();
        return rans;
    }

层序遍历（深度优先遍历）（利用队列实现）

/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     public int val;
 *     public TreeNode left;
 *     public TreeNode right;
 *     public TreeNode(int val=0, TreeNode left=null, TreeNode right=null) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
public class Solution {
    public IList<IList<int>> LevelOrder(TreeNode root) {
        List<IList<int>> ans = new();
        if (root == null) return ans;
        Queue<TreeNode> tmp = new();
        tmp.Enqueue(root);
        int size = 1;
        ans.Add(new List<int>{root.val});
        while (tmp.Count != 0){
            int i = size;
            size = 0;
            for (; i > 0; i--){

                TreeNode t = tmp.Dequeue();
                if (t.left != null){
                    tmp.Enqueue(t.left);
                    size++;
                } 
                if (t.right != null){
                    tmp.Enqueue(t.right);
                    size++;
                }
            }
            if (size != 0){
                List<int> a = new();
                foreach (var b in tmp.ToList()){
                    a.Add(b.val);
                }
                ans.Add(a);
            }
        }
        return ans;
    }
}
